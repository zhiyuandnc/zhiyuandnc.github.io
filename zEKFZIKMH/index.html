Graphics绘制LowB验证码

· 2021-01-06 · # Graphics # Java

一、创建ServletDemo继承HttpServlet，实现doGet和doPost方法

public class ServletDemo extends HttpServlet {
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        //画布宽度
        int width = 300;
        //画布高度
        int height = 100;
        //验证码长度
        int codeNum = 6;
        //干扰线数量
        int lineNum = 10;
        //验证码字符串
        String code = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
        //用于从验证码字符串中随机选取字符的验证码对象
        Random random = new Random();

        //创建RGB类型的内存画布
        BufferedImage bufferedImage = new BufferedImage(width,height,BufferedImage.TYPE_INT_RGB);
        //创建画笔
        Graphics graphics = bufferedImage.createGraphics();
        //设置画笔颜色
        graphics.setColor(Color.white);
        //填充背景色
        graphics.fillRect(0,0,width,height);
        //重设画笔颜色
        graphics.setColor(Color.red);
        //绘制边框
        graphics.drawRect(0,0,width-1,height-1);
        //设置画笔字体
        //通过字节码文件的相对路径取得字体文件的绝对路径，这里的字体文件是我ttf字体，你应该替换成你自己的
        String path = ServletDemo.class.getClassLoader().getResource("W03.ttf").getPath();
        Font font = null;
        if(new File(path).exists()){
            try {
                //从文件流中创建字体并设置字体大小为35和字体风格为加粗
                font = Font.createFont(Font.TRUETYPE_FONT, new FileInputStream(path)).deriveFont(35.0f).deriveFont(Font.BOLD);
            } catch (FontFormatException e) {
                e.printStackTrace();
            }
        }
        if(font!=null){
            graphics.setFont(font);
            //绘制文字
            for (int i = 0; i < codeNum; i++) {
                int pos = random.nextInt(62);
                int flag = 1;
                if(i%2!=0){
                    flag = -flag;
                }
                graphics.drawString(String.valueOf(code.charAt(pos)),i*50+5,height/2+flag*(10-i));
            }
        }
        //绘制干扰线
        graphics.setColor(Color.GRAY);
        for (int i = 0; i < lineNum; i++) {
            graphics.drawLine(width/(i+2)+20,(i+1)*10,width-i*10,height-i*10);
        }
        //绘制干扰点
        graphics.setColor(Color.BLUE);
        //输出图片到页面
        ImageIO.write(bufferedImage,"jpg",resp.getOutputStream());
    }

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        super.doGet(req, resp);
    }
}
